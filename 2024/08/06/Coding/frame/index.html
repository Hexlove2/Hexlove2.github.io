#define SAMPLERATE 16000
#define CHANNELS 2
#define BITDEEPTH 16

short frame1[4][320];

struct UserData{
    int     sampleRate      ;       // Sample rate
    int     channels        ;       // Number of channels
    short   bitDeepth       ;       // Bit depth
    short   frameDuration_In;       // Frame duration, in milliseconds
    short   frameDuration_P ;       // Frame duration, in milliseconds

    int           frameIndex;       // Frame index
    int           fileIndex ;       // Number of frames read from local file

    unsigned short        frame_size_In;   // Frame size
    unsigned short        frame_size_P ;   // Frame size

    unsigned short A_Process_Num       ;   // Frame loop index 1
    unsigned short B_Out_Num           ;   // Frame loop index 2
    unsigned short C_Processed_Nout_Num;   // Buffer 1
    unsigned short D_In_Unprocessed_Num;   // Buffer 2

    // Current data frame
    short        *current_On;
    short       *current_Mic;

    // Buffer
    short         *frame_PNO;
    short         *frame_INP;

    // Processed data position index
    int              P_index;

    // Store processed data
    short   *processed_Frame;

    // For testing purposes
    FILE             *output; // Output file
    int          count_write; // Total number of reads and writes
};

//*******************************************************************************************************
//*******************************************************************************************************

// Initialize userdata
void userData_Init(struct UserData *userdata, int sampleRate, int channels, short bitDeepth, short frameDuration, short frameDuration_P)
{
    // Initialize basic values
    userdata->sampleRate         =      sampleRate;
    userdata->channels           =        channels;
    userdata->bitDeepth          =       bitDeepth;
    userdata->frameDuration_In   =   frameDuration;
    userdata->frameDuration_P    = frameDuration_P;
    userdata->frameIndex         =               0;
    userdata->P_index            =               0;
    userdata->count_write        =               0;

    // Calculate frame size based on existing data
    userdata->frame_size_In      = (frameDuration*channels*sampleRate*bitDeepth)  /(8*1000);
    userdata->frame_size_P       = (frameDuration_P*channels*sampleRate*bitDeepth)/(8*1000);

    // Initialize recursive information
    userdata->A_Process_Num        = 0;
    userdata->B_Out_Num            = 0;
    userdata->C_Processed_Nout_Num = 0;
    userdata->D_In_Unprocessed_Num = 0;

    userdata->processed_Frame    = (short *)malloc(userdata->frame_size_In)         ;
    userdata->frame_INP          = (short *)malloc(userdata->frame_size_P)          ;
    userdata->frame_PNO          = (short *)malloc(userdata->frame_size_In)         ;
}

// Free userdata space
void userData_Free(struct UserData *userdata)
{
    free(userdata->processed_Frame);
}

// Data merging
void buffer_Merge(short *buffer1, int size1, short *buffer2, int size2, short *output)
{
    for(int i=0; i< (size1 + size2)/2; i++)
    {
        if(i < size1 / 2)
        {
            output[i] = buffer1[i];
        }
        else
        {
            output[i] = buffer2[i-size1 / 2];
        }
    }
}

// Channel separation
void frame_Read(short *buffer1, short *buffer2, short *out, short size)
{
    for(int i =0; i < size / 4 ; i++)
    {
        out[2 * i   ] = buffer1[2*i + 1];
        out[2 * i +1] = buffer2[2*i    ];
    }
}

// The input data format in_size must be delivered in a 16ms format, and the function's return value out_size is the length of the output mono data frame.

int preprocess_short(void *st_ptr, short *in, short *out, int in_size, int *wakeup_status)
{
    int out_size;

    // Data processing
    memcpy(out, in , in_size);

    return 1;
}

// AudioRendererOnWriteData represents the audio playback data of the device, with audio format of 16000 sample rate, stereo, 16-bit, 20ms per frame.
// buffer is part of the data playback callback, bufferLen is the length of the data frame. The stereo data content is the same.

static int32_t AudioRendererOnWriteData(void *userData, void *buffer, int32_t bufferLen)
{
    // Read frame data from buffer into userData
    short *bufferData = (short *)(buffer);
    struct UserData *userdata   = (struct UserData *)userData;

    // Ensure it's a frame length, in bytes. If the incoming frame data size is incorrect, return error information.
    if(bufferLen != userdata->frame_size_In) return -2;
    
    // Update the current frame address
    userdata->current_On   = bufferData;

    return 1;
}

// AudioCapturerOnReadData represents the audio data captured by the device, with audio format of 16000 sample rate, stereo, 16-bit, 20ms per frame.
// buffer is part of the data capture callback, bufferLen is the length of the data frame. Only the right channel of the stereo is mic-collected data.
// Now, within the AudioCapturerOnReadData function, we need to implement the following:
// Send the data received by AudioRendererOnWriteData and the data captured by AudioCapturerOnReadData to preprocess_short algorithm for processing.
// After processing, convert the processed data to the format of 16000 sample rate, mono, 16-bit, 20ms per frame.

// Each time a frame of data is captured, this function is automatically called.
static int32_t AudioCapturerOnReadData(void *userData, void *buffer, int32_t bufferLen) 
{    
    // Read frame data from buffer into userData
    short  *bufferData = (short *)(buffer);
    struct UserData *userdata   = (struct UserData *)userData;
    int    result;

    // Ensure it's a frame length
    if(bufferLen != userdata->frame_size_In) return -2;

    // Current and previous frames
    short *current_On    = userdata->current_On   ;
    short *current_Mic   = userdata->current_Mic  ;
    int   *frameIndex    = &userdata->frameIndex  ;

    // Read the current frame
    current_Mic  = bufferData;

    // Current frame
    short *frame = (short *)malloc(userdata->frame_size_In);

    // Combine local and microphone frame data
    frame_Read(current_On, current_Mic, frame, userdata->frame_size_In);

    // Update the required information for this round
    unsigned short *A_N = &userdata->A_Process_Num       ;
    unsigned short *B_N = &userdata->B_Out_Num           ;
    unsigned short *C_N = &userdata->C_Processed_Nout_Num;
    unsigned short *D_N = &userdata->D_In_Unprocessed_Num;

    short *buffer1      = (short *)malloc(userdata->frame_size_In + userdata->frame_size_P);
    short *buffer2      = (short *)malloc(sizeof(short) * userdata->frame_size_In);
    short *buffer3      = (short *)malloc(sizeof(short) * userdata->frame_size_In);

    // Merge buffers
    buffer_Merge(userdata->frame_INP, *D_N, frame, userdata->frame_size_In, buffer1);

    short C_LN = *C_N;
    short D_LN = *D_N;

    // Calculate information values
    // Number of processing loops
    *A_N = (userdata->frame_size_In + *D_N) / (userdata->frame_size_P);
    // Number of output frames
    *B_N = ((*C_N) + userdata->frame_size_P * (*A_N)) / (userdata->frame_size_In);
    // Number of bytes of processed but not output frame data
    *C_N = *C_N + (*A_N) * userdata->frame_size_P - (*B_N) * userdata->frame_size_In;
    // Number of bytes of input but not processed data
    *D_N = *D_N + userdata->frame_size_In - (*A_N) * userdata->frame_size_P;

    short *buffer_P_In  = (short *)malloc(userdata->frame_size_P);
    short *buffer_P_Out = (short *)malloc(userdata->frame_size_P);
    short *buffer_O     = (short *)malloc(userdata->frame_size_In); 
    
    // Frame processing loop
    for(int i = 0; i < *A_N; i++)
    {
    // Read the merged buffer1
    for(int j = 0; j < userdata->frame_size_P / 2; j++)
    {
    buffer_P_In[j] = buffer1[i * userdata->frame_size_P / 2 + j];
    }
          // Frame processing function
    preprocess_short(NULL, buffer_P_In, buffer_P_Out, userdata->frame_size_P, 0);

    // Put the processed data into the buffer
    for(int j = 0; j < userdata->frame_size_P / 2; j++)
    {
        buffer3[i * userdata->frame_size_P / 2 + j ] = buffer_P_Out[j];
    }

    // Merge processed data frames
    buffer_Merge(userdata->frame_PNO, C_LN, buffer3, (*A_N) * userdata->frame_size_P, buffer2);

    // Loop output frames, up to two frames
    for(int i = 0; i < *B_N; i++)
    {
        for(int j = 0; j < userdata->frame_size_In / 2; j++)
        {
            buffer_O[j] = buffer2[i * userdata->frame_size_In / 2 + j];
        }
        // Frame output completed, please send out frame information in a timely manner
        fwrite(buffer_O, 2, userdata->frame_size_In / 2, userdata->output); // For testing
    }

    // Update two buffers
    // Unprocessed part in buffer1
    for(int i = 0; i < *D_N / 2; i++)
    {
        userdata->frame_INP[i] = buffer1[(*A_N)*userdata->frame_size_P/2 + i];
    }    
    // Unsent part in buffer2
    for(int i = 0; i < *C_N / 2; i++)
    {
        userdata->frame_PNO[i] = buffer2[(*B_N)*userdata->frame_size_In/2 +i];
    }

    // Free the allocated heap space
    free(buffer1);
    free(buffer2);
    free(buffer3);
    free(buffer_O);
    free(buffer_P_In);
    free(buffer_P_Out);
    free(frame);

    // Set pointers to NULL to prevent wild pointers
    buffer1      = NULL;
    buffer2      = NULL;
    buffer3      = NULL;
    buffer_O     = NULL;
    buffer_P_In  = NULL;
    buffer_P_Out = NULL;
    frame        = NULL;
    return 1;
