# Image Processing Algorithm(Raw Data)

Supposing that we have an AVFrame constructor point, like *frame

void frame_horizontal(AVFrame* frame)
{
	int width = frame->width;
	int height = frame->height;
	int uv_width = width / 2;
	int uv_height = height / 2;

	for (int i = 0; i < height; i++)
	{
		uint8_t * row = frame->data[0] + i * (frame->linesize[0]);
		for (int j = 0; j < width / 2; j++)
		{
			uint8_t temp;
			temp = row[j];
			row[j] = row[width - 1 - j];
			row[width - 1 - j] = temp;
		}
	}

	for (int i = 0; i < uv_height; i++)
	{
		uint8_t * row_u = frame->data[1] + i * (frame->linesize[1]);
		uint8_t * row_v = frame->data[2] + i * (frame->linesize[2]);
		for (int j = 0; j < uv_width / 2; j++)
		{
			uint8_t temp;
			temp = row_u[j];
			row_u[j] = row_u[uv_width - j - 1];
			row_u[uv_width - j - 1] = temp;

			temp = row_v[j];
			row_v[j] = row_v[uv_width - j - 1];
			row_v[uv_width - j - 1] = temp;
		}
	}
}

static void frame_vertical(AVFrame* frame)
{
	int height = frame->height;
	int width = frame->width;
	int uv_height = height / 2;
	int uv_width = width / 2;

	for (int i = 0; i < height / 2; i++)
	{
		uint8_t * row_1 = frame->data[0] + i * (frame->linesize[0]);
		uint8_t * row_2 = frame->data[0] + (height - i - 1) * (frame->linesize[0]);
		for (int j = 0; j < width; j++)
		{
			int temp = row_1[j];
			row_1[j] = row_2[j];
			row_2[j] = temp;
		}
	}

	for (int i = 0; i < uv_height / 2; i++)
	{
		uint8_t* row_u_1 = frame->data[1] + i * (frame->linesize[1]);
		uint8_t* row_u_2 = frame->data[1] + (uv_height - i - 1) * (frame->linesize[1]);
		uint8_t* row_v_1 = frame->data[2] + i * (frame->linesize[2]);
		uint8_t* row_v_2 = frame->data[2] + (uv_height - i - 1) * (frame->linesize[2]);
		for (int j = 0; j < uv_width; j++)
		{
			int temp = row_u_1[j];
			row_u_1[j] = row_u_2[j];
			row_u_2[j] = temp;

			temp = row_v_1[j];
			row_v_1[j] = row_v_2[j];
			row_v_2[j] = temp;
		}
	}
}

static void frame_gray(AVFrame* frame)
{
	int height = frame->height / 2;
	int width = frame->width / 2;
	for (int i = 0; i < height; i++)
	{
		uint8_t* row_u = frame->data[1] + i * (frame->linesize[1]);
		uint8_t* row_v = frame->data[2] + i * (frame->linesize[2]);
		for (int j = 0; j < width; j++)
		{
			row_u[j] = 128;
			row_v[j] = 128;
		}
	}
}

static void frame_auto_light(AVFrame* frame)
{
	if (frame->data == NULL) return;

	int height = frame->height;
	int width = frame->width;
	int count = 0;
	for (int i = 0; i < height; i++)
	{
		uint8_t* row = frame->data[0] + i * (frame->linesize[0]);
		for (int j = 0; j < width; j++)
		{
			if (row[j] > 140)
				count++;
		}
	}
	float index_light = 2.4f - ((float)count / (float)(height * width)) * 3.5f;
	SLOG_ERROR("Current frame index of light : %f percent:%f count:%d\n", index_light, ((float)count / (float)(height * width)), count);
	if (index_light > 2.2f)
		index_light = 2.2f;
	else if (index_light < 0.0f)
		index_light = 0.3f;
	frame_gamma_adjust(frame, index_light);
}

void gaussian_create_kernel(double* kernal, double sigma, int kernal_size)
{
	if (kernal_size % 2 == 0) kernal_size--;
	const double pi = 3.141592653589793;
	double sum = 0.0;

	//SLOG_ERROR("Kernal Size:%d\n", kernal_size);
	//kernal = (double*)malloc(sizeof(double) * kernal_size * kernal_size);

	int mid = (kernal_size - 1) / 2;

	for (int i = 0; i < kernal_size; i++)
	{
		for (int j = 0; j < kernal_size; j++)
		{
			int x = i - mid;
			int y = j - mid;
			double value = (1.0 / (pi * 2.0 * sigma * sigma)) * exp((-1.0)*(x * x + y * y) / (2.0 * sigma * sigma));
			*(kernal + j + i * kernal_size) = value;
			sum += value;
		}
	}

	//πÈ“ªªØ
	SLOG_ERROR("Kernal Display:%f",sum);
	sum = 1 / sum;
	//SLOG_ERROR("Kernal Display:");
	for (int i = 0; i < kernal_size; i++)
	{
		//SLOG_ERROR("||");
		for (int j = 0; j < kernal_size; j++)
		{
			*(kernal + j + i * kernal_size) *= sum;
			//SLOG_ERROR("%f  ", *(kernal + j + i * kernal_size));
		}
		//SLOG_ERROR("||\n");
	}
	//SLOG_ERROR("gaussian_create_kernel success");
}

static void BGR2YUV(uint8_t *data)
{
	uint8_t* newdata = (uint8_t*)malloc(1920 * 1080 * 3);
	//memcpy(newdata, data, 1920 * 1080 * 3);
	for (int i = 0; i < 1920 * 1080; i++)
	{
		uint8_t B = data[i * 3];
		uint8_t G = data[i * 3 + 1];
		uint8_t R = data[i * 3 + 2];

		uint8_t Y = 0.2126*R + 0.7152*G + 0.0722*B;
		uint8_t U = -0.09991*R  - 0.33609*G + 0.436*B;
		uint8_t V = 0.615*R - 0.55861*G-0.05639*B ;

		newdata[i] = Y;
		newdata[i + 1920 * 1080] = U;
		newdata[i + 2 * 1920 * 1080] = V;
	}
	memcpy(data, newdata, 1920 * 1080 * 3);
	free(newdata);
}

static void YUV2BGR(uint8_t* data)
{
	uint8_t* newdata = (uint8_t*)malloc(1920 * 1080 * 3);
	//memcpy(newdata, data, 1920 * 1080 * 3);
	for (int i = 0; i < 1920 * 1080; i++)
	{
		uint8_t Y = data[i];
		uint8_t U = data[i + 1920 * 1080];
		uint8_t V = data[i + 2 * 1920 * 1080];

		uint8_t B = Y + 2.12798 * U;
		uint8_t G = Y - 0.21482 * U - 0.38059 * V;
		uint8_t R = Y + 1.28033 * V;

		newdata[3*i] = R;
		newdata[3*i + 1] = G;
		newdata[3*i + 2] = B;
	}
	memcpy(data, newdata, 1920 * 1080 * 3);
	free(newdata);
}


void frame_DGamma_adjust(int kernal_size, double *kernal_1, double* kernal_2, double* kernal_3, void *frame1, int type)
{
	int width;
	int height;
	int linesize;
	uint8_t* data;
	if (type == 0)
	{
		AVFrame* frame = (AVFrame*)frame1;
		data = frame->data[0];
		width = frame->width;
		height = frame->height;
		linesize = frame->linesize[0];
	}

	else
	{
		width = 1920;
		height = 1080;
		linesize = 1920;
		data = (uint8_t*)frame1;
		BGR2YUV(data);
	}
		//SLOG_ERROR("frame_2DGamma Start1 \n");

		
		//SLOG_ERROR("frame_2DGamma Start2 \n");
		double* light = (double*)malloc(sizeof(double) * 1920 * 1080);
		memset(light, 0, sizeof(double) * 1920 * 1080);
		double mean = 0;
		//SLOG_ERROR("frame_2DGamma Start3 \n");

		for (int i = 0; i < height; i++)
		{
			for (int j = 0; j < width; j++)
			{
				double sum = 0.0;
				for (int q = 0; q < kernal_size; q++)
				{
					for (int w = 0; w < kernal_size; w++)
					{
						int x = i + q - 2;
						int y = j + w - 2;
						if (x < 0) x = 0;
						if (x >= height) x = height - 1;
						if (y < 0) y = 0;
						if (y >= width)  y = width - 1;
						double a1 = kernal_1[w + kernal_size * q];
						double a2 = kernal_2[w + kernal_size * q];
						double a3 = kernal_3[w + kernal_size * q];
						double b1 = (double)(data[y + x * width]);
						sum += ((a1 + a2 + a3) * b1) / 3;
					}
				}
				light[j + i * 1920] = sum;
				mean += sum;
			}
		}



		mean = mean / (double)(1920 * 1080);
		SLOG_ERROR("Mean Display:%f", mean);

		for (int i = 0; i < height; i++)
		{
			for (int j = 0; j < width; j++)
			{
				double r = pow(0.5, (mean - light[j + i * 1920]) / mean);
				double F = (double)(data[j + i * width]);

				//SLOG_ERROR("frame_2DGamma Start4 \n");
				uint8_t value = (uint8_t)(255.0 * pow(F / 255.0, r));
				//SLOG_ERROR("frame_2DGamma Start5 \n");
				if (value < 0) value = 0;
				else if (value > 255) value = 255;
				data[j + i * width] = value;
			}
		}
	
		if (type == 1) YUV2BGR(data);
}

static void frame_histogram(AVFrame* frame)
{
	int width = frame->width;
	int height = frame->height;
	int linesize = frame->linesize[0];
	uint8_t* data = frame->data[0];
	int counts[256];
	uint8_t y[256];
	int sum = 0;

	memset(counts, 0, 256 * 4);
	memset(y, 0, 256);

	for (int i = 0; i < height; i++)
	{
		uint8_t* row = frame->data[0] + i * (frame->linesize[0]);
		for (int j = 0; j < width; j++)
		{
			counts[row[j]]++;
		}
	}

	for (int i = 0; i < 256; i++)
	{
		sum += counts[i];
		int x;
		x = 255 * (sum) / (height * width);
		y[i] = x;
		SLOG_ERROR("Y[%d]->%d \n",i, x);
	}

	for (int i = 0; i < height; i++)
	{
		uint8_t* row = frame->data[0] + i * (frame->linesize[0]);
		for (int j = 0; j < width; j++)
		{
			row[j] = y[row[j]];
		}
	}
}
static void frame_mean_blur(AVFrame* frame)
{
	int width = frame->width;
	int height = frame->height;
	int linesize = frame->linesize[0];
	uint8_t* data = frame->data[0];

	float kernal[3][3] = {
		{1.0f / 9, 1.0f / 9, 1.0f / 9 },
		{1.0f / 9, 1.0f / 9, 1.0f / 9 },
		{1.0f / 9, 1.0f / 9, 1.0f / 9 }
	};

	uint8_t* output = malloc(height * linesize);
	memcpy(output, data, height * linesize);
	int pad = 3 / 2;
	for(int i = pad; i < height - pad; i++)
		for (int j = pad; j < width - pad; j++)
		{
			float sum = 0.0f;
			for (int x = 0; x < 3; x++)
			{
				for (int y = 0; y < 3; y++)
				{
					sum += output[j - 1 + y + (i - 1 + x) * linesize] * kernal[x][y];
				}
			}
			data[j + i * linesize] = (uint8_t)sum;
		}
	free(output);
}

static void frame_sharpening(AVFrame* frame)
{
	int width = frame->width;
	int height = frame->height;
	int linesize = frame->linesize[0];
	uint8_t* data = frame->data[0];

	float kernal[3][3] = {
		{-1.0f, -1.0f, -1.0f},
		{-1.0f,  9.0f, -1.0f},
		{-1.0f, -1.0f, -1.0f}
	};

	uint8_t* output = malloc(height * linesize);
	memcpy(output, data, height * linesize);
	int pad = 3 / 2;
	for (int i = pad; i < height - pad; i++)
		for (int j = pad; j < width - pad; j++)
		{
			float sum = 0.0f;
			for (int x = 0; x < 3; x++)
			{
				for (int y = 0; y < 3; y++)
				{
					sum += output[j - 1 + y + (i - 1 + x) * linesize] * kernal[x][y];
				}
			}
			if (sum < 0)
			{
				data[j + i * linesize] = 0;
			}
			else if (sum > 255)
			{
				data[j + i * linesize] = 255;
			}
			else
				data[j + i * linesize] = (uint8_t)sum;
		}
	free(output);
}

static void frame_kernal(AVFrame* frame, float * kernal)
{
	int width = frame->width;
	int height = frame->height;
	int linesize = frame->linesize[0];
	uint8_t* data = frame->data[0];


	uint8_t* output = malloc(height * linesize);
	memcpy(output, data, height * linesize);
	int pad = 3 / 2;
	for (int i = pad; i < height - pad; i++)
		for (int j = pad; j < width - pad; j++)
		{
			float sum = 0.0f;
			for (int x = 0; x < 3; x++)
			{
				for (int y = 0; y < 3; y++)
				{
					sum += output[j - 1 + y + (i - 1 + x) * linesize] * (*(kernal+y+3*x));
				}
			}
			if (sum < 0)
			{
				data[j + i * linesize] = 0;
			}
			else if (sum > 255)
			{
				data[j + i * linesize] = 255;
			}
			else
				data[j + i * linesize] = (uint8_t)sum;
		}
	free(output);
}

static void frame_gamma_adjust(AVFrame *frame, float gamma)
{
	int height = frame->height;
	int width = frame->width;
	int linesize = frame->linesize[0];
	uint8_t* data = frame->data[0];
	
	for (int i = 0; i < height; i++)
	{
		uint8_t* row = data + i * linesize;
		for (int j = 0; j < width; j++)
		{
			float a = (row[j]) / 255.0f;
			float b = pow(a, 1.0f / gamma);
			a = (b * 255.0f + 0.5f);
			if (a > 255.0f)
			{
				row[j] = 255;
			}
			else
				row[j] = (uint8_t)a;
			
		}
	}
}
