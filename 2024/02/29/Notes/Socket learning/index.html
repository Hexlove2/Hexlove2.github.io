# 5.TCP 原理

# 1. 缓冲区

  • I/O 缓冲在每个 tcp 套接字中单独存在
  • I/O 缓冲在创建套接字时自动生成
  • 即使关闭套接字也会继续传递输出缓冲中遗留的数据
  • 关闭套接字将丢失输入缓冲区中的数据

# 6.UDP

# 1. 一般特性

  • UDP 传输中，双方调用 I/O 次数要一致
  • 如果不提前连接好套接字的目标地址端口，那么每次使用 sendto 函数时都会反复注册与删除目标的地址信息，十分耗费时间，长时间使用 udp 最好提前注册好

# 2.SO_REUSEADDR 地址再分配

主动断开连接的一方一般会有个 time-wait 时间，并且是基于套接字的，因为要防止最后的 FIN 消息确认 ACK 成功送达对方，否则对方会一直重复发送 FIN 消息，直接到达重复次数限制主动断开。如果是客户端那还好，因为客户端可以一直动态分配端口号，用的套接字也不一样，但服务器端只有那么多端口号，因为 ip 变不了，在网络较差的环境下 time-wait 被重复触发可能会造成端口号耗尽，因此需要一定的优化。使用也比较方便，设为 1 即可允许再分配。

int option = true;
optlen = sizeof(int);
state = setsockopt(serv_sock, SOL_SOCKET, SO_REUSEADDR, (void*)&option, optlen);

signal (信号类型，发生时调用的函数名)

#include <stdio.h>
#include <unistd.h>
#include <signal.h>

//必须是void func(int sig)
void timeout(int sig)
{
    if(sig==SIGALRM) //说明是alarm导致
    puts("Time out!");
    alarm(2);//这里面会导致重复触发
}

void keycontrol(int sig)
{
    if(sig==SIGINT) //键盘输入了ctrl+c
    puts("ctrl+c pressed");
}

int main(int argc, char *argv[])
{
    int i;
    //绑定好信号量对应的函数
    signal(SIGALRM,timeout);  
    signal(SIGINT,keycontrol);

    alarm(2);//2秒后触发信号量

    for(i=0;i<3;i++)
    {
        puts("Wait....");
        //表面上看是100秒睡眠，但由于上面的2秒就触发了，导致系统为了执行本进程的相关函数
        //强行唤醒了阻塞的程序，也就是说sleep不到2秒就被唤醒了
        sleep(100);
    }
    return 0;
}

sigaction 函数

上面的 plus 版，更加通用兼容强大

仅展示如何使用

#include <stdio.h>
#include <unistd.h>
#include <signal.h>

//必须是void func(int sig)
void timeout(int sig)
{
    if(sig==SIGALRM) //说明是alarm导致
    puts("Time out!");
    alarm(2);//这里面会导致重复触发
}

int main(int argc, char *argv[])
{
    int i;
    //act初始化
    struct sigaction act;
    act.sa_handler = timeout;
    sigemptyset(&act.sa_mask);
    act.sa_flags=0;

    //调用
    sigaction(SIGALRM,&act,0);
    
    alarm(2);//2秒后触发信号量

    for(i=0;i<3;i++)
    {
        puts("Wait....");
        sleep(100);
    }
    return 0;
}

结合以上所有内容，实现多进程并发回声服务器

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <signal.h>
#include <arpa/inet.h>
#include <sys/wait.h>
#include <sys/socket.h>

#define BUF_SIZE 30
void error_handling(char *message);
void read_childproc(int sig);

int main(int argc, char const *argv[])
{
    //服务器socket相关
    int serv_sock, clnt_sock;
    struct sockaddr_in serv_adr, clnt_adr;
    socklen_t adr_sz;

    char buf[BUF_SIZE];
    //多进程相关
    pid_t pid;
    struct sigaction act;
    int str_len, state;
    
    //初始化完成
    if(argc!=2)
     {
        printf("Usage:%s <port> \n", argv[0]);
        exit(1);
     }

    //多进程初始化
    act.sa_handler=read_childproc;
    act.sa_flags=0;
    sigemptyset(&act.sa_mask);

    //绑定信号,当子进程结束时调用
    state=sigaction(SIGCHLD, &act, 0);

    //服务器套接字初始化
    serv_sock=socket(PF_INET, SOCK_STREAM, 0);
    memset(&serv_adr, 0, sizeof(serv_adr));
    serv_adr.sin_family=AF_INET;
    serv_adr.sin_addr.s_addr=htonl(INADDR_ANY);
    serv_adr.sin_port=htons(atoi(argv[1]));

    //绑定
    if(bind(serv_sock, (struct sockaddr*)&serv_adr, sizeof(serv_adr))==-1)
     error_handling("bind error");
    
    //监听
    if(listen(serv_sock, 5)==-1)
     error_handling("listen error");

    //多进程服务
    while(1)
    {
        //客户端套接字创建
        adr_sz=sizeof(clnt_adr);
        clnt_sock=accept(serv_sock, (struct sockaddr*)&clnt_adr, &adr_sz);

        //多进程时，要满足多个客户端
        if(clnt_sock==-1)
         continue;
        else
         puts("new client connected");

        //开启多进程
        //父进程pid为本身，而子进程的pid则为0
        pid=fork();

        //假设进程创建失败
        if(pid==-1)
        {
            close(clnt_sock);
            continue;
        }
        //子进程运行区域
        if(pid==0)
        {
            //此时可关闭服务器初始套接字,不会影响父进程中的套接字
            close(serv_sock);

            //服务客户端
            //echo,不过不是按字节读取，当读取内容过大时可能造成问题
            while((str_len=read(clnt_sock, buf, BUF_SIZE))!=0)
                write(clnt_sock, buf, str_len);
            
            //关闭套接字
            close(clnt_sock);
            puts("client disconnected");

            //关闭子进程，且会调用子进程结束函数,不会出现僵尸进程
            return 0;
        }

        //父进程运行区域
        else
        {
            //因为此时子进程已经复制了,可以直接关掉,只有当所有进程中的文件描述符均关闭后，套接字才会消失
            close(clnt_sock);
        }
    }

    //关闭服务器套接字
    close(serv_sock);
    return 0;
}

void error_handling(char *message)
{
    fputs(message, stderr);
    fputc('\n', stderr);
    exit(1);
}

void read_childproc(int sig)
{
    pid_t pid;
    int status;
    //不会造成阻塞，且此时子进程一定已结束
    pid=waitpid(-1, &status, WNOHANG);
    printf("removed proc id :%d \n",pid);
}

# 5. 分割 TCP 的 I/O 程序

例如让父进程专门接受数据，让子进程专门发送数据，而不会出现，还没接受到，就一直不发送的情况，提高效率。

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <sys/socket.h>

#define BUF_SIZE 30
void error_handling(char *message);
void read_routine(int sock, char *buf);
void write_routine(int sock, char *buf);

int main(int argc, char const *argv[])
{
    int sock;
    struct sockaddr_in serv_adr;
    pid_t pid;
    char buf[BUF_SIZE];

    sock=socket(PF_INET,SOCK_STREAM,0);
    memset(&serv_adr, 0, sizeof(serv_adr));
    serv_adr.sin_family=AF_INET;
    serv_adr.sin_addr.s_addr=inet_addr(argv[1]);
    serv_adr.sin_port=htons(atoi(argv[2]));

    if(connect(sock, (struct sockaddr*)&serv_adr, sizeof(serv_adr))==-1)
        error_handling("connect error");
    
    pid=fork();
    if(pid==0)
    {
        fputs("Input Message(Q to quit) \n", stdout);
        write_routine(sock, buf);
    }
    else
        read_routine(sock, buf);
    close(sock);
    return 0;
}

void read_routine(int sock, char *buf)
{
    while(1)
    {
        int str_len=read(sock, buf, BUF_SIZE);
        if(str_len==0)
         return;
        
        buf[str_len]=0;
        printf("Message from server: %s",buf);
    }
}

void write_routine(int sock, char *buf)
{
    while(1)
    {
        fgets(buf, BUF_SIZE, stdin);
        if(!strcmp(buf, "q\n")||!strcmp(buf, "Q\n"))
        {
            shutdown(sock, SHUT_WR);
            return;
        }
        write(sock, buf, strlen(buf));
    }
}

void error_handling(char *message)
{
    fputs(message, stderr);
    fputc('\n', stderr);
    exit(1);
}

# 11. 进程间的通信

#include <unistd.h>

int pipe(int filedes[2];

单管道虽然可以实现双向通信，但很容易产生很多错误，因此最好创建两个管道以实现双向通信

#include <stdio.h>
#include <unistd.h>
#define BUF_SIZE 30
int main(int argc, char const *argv[])
{
    pid_t pid;
    //create two pipe
    int fds1[2],fds2[2];
    char mes1[]="Who are you?";
    char mes2[]="I am your son.";
    char buf[BUF_SIZE];

    pipe(fds1);
    pipe(fds2);

    pid=fork();
    if(pid==0)
    {
        write(fds1[1], mes1, sizeof(mes1));
        read(fds2[0], buf, BUF_SIZE);
        printf("Child proc output: %s \\n", buf);
    }
    else
    {
        read(fds1[0], buf, BUF_SIZE);
        printf("Parent proc output: %s \\n", buf);
        write(fds2[1], mes2, sizeof(mes2));
        sleep(3);
    }

    return 0;
}

#include <unistd.h>
#include <string.h>
#include <signal.h>
#include <sys/wait.h>

{
    //about sock
    int serv_sock, clnt_sock;
    struct sockaddr_in serv_adr, clnt_adr;
    socklen_t adr_sz;

    //pid
    pid_t pid;

    //pipe
    int fds[2];

    //signal
    struct sigaction act;

    //other
    char buf[BUF_SIZE];
    int state, str_len;

    if(argc!=2)
    {
        printf("Usage: %s <port>",argv[0]);
        exit(1);
    }

    //signal set
    act.sa_handler=read_childproc;
    sigemptyset(&act.sa_mask);
    act.sa_flags=0;
    state=sigaction(SIGCHLD, &act, 0);

    //socket set
    serv_sock=socket(PF_INET, SOCK_STREAM, 0);
    memset(&serv_adr, 0, sizeof(serv_adr));
    serv_adr.sin_family=AF_INET;
    serv_adr.sin_addr.s_addr=htonl(INADDR_ANY);
    serv_adr.sin_port=htons(atoi(argv[1]));

    if(bind(serv_sock,(struct sockaddr*)&serv_adr,sizeof(serv_adr))==-1)
     error_handling("bind error");
    
    if(listen(serv_sock,5)==-1)
     error_handling("listen error");
    
    //using pipe to communicate with father process
    pipe(fds);
    //create a new process to write message to file
    pid=fork();
    //son process
    if(pid==0)
    {
        FILE * fp = fopen("echomsg.txt","wt");
        char msgbuf[BUF_SIZE];
        int i,len;
        for(i=0;i<10;i++)
        {
            len=read(fds[0],msgbuf,BUF_SIZE);
            //printf("form write pipe:%s",msgbuf);
            fwrite((void *)msgbuf,1,len,fp);
        }
        fclose(fp);
        return 0;
    }

    //echo multiprocessing
    while(1)
    {
        adr_sz=sizeof(clnt_adr);
        clnt_sock=accept(serv_sock,(struct sockaddr*)&clnt_adr,&adr_sz);
        if(clnt_sock==-1)
            continue;
        else
            puts("new client connected");
        
        pid=fork();
        if(pid==0)
        {
            close(serv_sock);

                write(clnt_sock,buf,str_len);
                //to pipe
                write(fds[1],buf, str_len);
            }
            close(clnt_sock);
            puts("client disconnected...");
            return 0;
        }
        else
        {
            close(clnt_sock);
        }
    }
    close(serv_sock);
    return 0;
}

# 1.select 函数

将多个文件描述符统一起来集中管理

使用步骤：

1. 设置文件描述符，指定监视范围，设置超时

2. 调用 select 函数

3. 查看调用结果

fd_set 结构：

int main(int argc, char *argv[])
{
	fd_set set;
	FD_ZERO(&set);//变量初始化为0，每次用建议都加上
	FD_SET(fd,&set);//将文件描述符为fd的位置置1，表示加入监视范围
	//0为stdin，1为stdout，2为stderr
	FD_SET(1,&set);
	FD_CLR(fd,&set);//fd位置0
}

使用实例：

#include <stdio.h>
#include <unistd.h>
#include <sys/time.h>
#include <sys/select.h>
#define BUF_SIZE 30

int main(int argc, char *argv[])
{
    int result,str_len;
    fd_set reads, temps;
    char buf[BUF_SIZE];
    struct timeval timeout;

    FD_ZERO(&reads);
    FD_SET(0, &reads);

    while(1)
    {
        temps=reads;
        timeout.tv_sec=5;
        timeout.tv_usec=0;
        result=select(1,&temps,0,0,&timeout);
        if(result==-1)
        {
            puts("select error");
            break;
        }
        else if(result==0)
         puts("Time out!");
        else 
        {
            if(FD_ISSET(0, &temps))
                {
                    str_len=read(0, buf, BUF_SIZE);
                    buf[str_len]=0;
                    printf("Message from console: %s",buf);
                }    
        }
    }
    return 0;
}

# 2. 实现 I/O 复用服务器端

代码如下

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <arpa/inet.h>
#include <sys/socket.h>
#include <sys/select.h>
#include <sys/time.h>

#define BUF_SIZE 100
void error_handling(char *message);

int main(int argc, char const *argv[])
{
    //socket
    int serv_sock, clnt_sock;
    struct sockaddr_in serv_adr,clnt_adr;
    socklen_t adr_sz;

    //select
    fd_set reads,cpy_reads;
    struct timeval timeout;

    //other
    int i, str_len,fd_max,fd_num;
    char buf[BUF_SIZE];
    if(argc!=2)
    {
        printf("Usage: %s <port>", argv[0]);
        exit(1);
    }

    serv_sock=socket(PF_INET,SOCK_STREAM,0);
    memset(&serv_adr,0,sizeof(serv_adr));
    serv_adr.sin_family=AF_INET;
    serv_adr.sin_addr.s_addr=htonl(INADDR_ANY);
    serv_adr.sin_port=htons(atoi(argv[1]));

    if(bind(serv_sock, (struct sockaddr*)&serv_adr, sizeof(serv_adr))==-1)
     error_handling("bind error");
    
    if(listen(serv_sock,5)==-1)
     error_handling("listen error");
    
    FD_ZERO(&reads);
    FD_SET(serv_sock,&reads);
    fd_max=serv_sock;

    while(1)
    {
        cpy_reads=reads;
        timeout.tv_sec=5;
        timeout.tv_usec=5000;

        //error
        if((fd_num=select(fd_max+1,&cpy_reads,0,0,&timeout))==-1)
         break;
        //no data in
        else if(fd_num==0)
         continue;

        //detect every file
        for(i=0;i<fd_max+1;i++)
        {
            //new client are connecting
            if(FD_ISSET(i,&cpy_reads))
            {
                if(i==serv_sock)
                {
                    adr_sz=sizeof(clnt_adr);
                    clnt_sock=accept(serv_sock,(struct sockaddr*)&clnt_adr,&adr_sz);
                    //update to reads but not cpy_reads
                    FD_SET(clnt_sock,&reads);
                    if(clnt_sock>fd_max)
                     fd_max=clnt_sock;
                    printf("connected client:%d \\n" ,clnt_sock);
                }
                //client are texting
                else
                {
                    str_len=read(i,buf,BUF_SIZE);
                    //Client quit
                    if(str_len==0)
                    {
                        FD_CLR(i,&reads);
                        close(i);
                        printf("closed client:%d \\n",i);
                    }

                    //echo
                    else
                    {
                        write(i,buf,str_len);
                    }
                }

            }
        }
    }
    close(serv_sock);
    return 0;
}

void error_handling(char *message)
{
    fputs(message, stderr);
    fputc('\\n',stderr);
    exit(1);
}

首先接受端在收到紧急信息时会触发一个紧急信号，STGURG，也就是说，可以用信号量机制来处理，但问题是，如果有多个进程，例如子进程，都能收到这个信号，谁去处理呢？

方法是用 fcntl 函数指定一个进程 id，去处理这个紧急消息

# 15. 套接字和标准 I/O

# 1. 标准 IO 函数的优缺点

1. 具有良好的移植性

2. 可利用缓冲提高性能

由于数据包中首部的存在，缓冲可以有效减少数据包的发送数量，效率更高，标准 IO 函数会有单独的数据缓冲区，而创建套接字时，也会生成套接字专用缓冲区，一共两层，套接字缓冲最接近实际网络。

若想实现一个复制文件的操作，并且是一个大小不小的文件，例如 300MB 的一个 txt

这时使用 read&write 函数，则需数分钟以上

如果没缺点谁还不用这个呢

1. 不容易进行双向通信

2. 有时频繁调用 fflush 函数

3. 需要以 FILE 结构体指针的形式返还文件描述符

平常的文件描述符，就是个 int 型的数，但标准 IO 函数必须用 FILE 格式的指针，当然这是可以转化的

# 16.I/O 流分离

# 1. 分离 IO 流

之前的两次 IO 分离

1. 利用 fork ( ) 函数实现，子进程与父进程分别 IO

2.FILE 读模式指针和写模式指针

优点：

1. 可以分开去编写输入和输出函数，易于开发维护

2. 与输入无关的输出操作速度得以提高

3. 区分 IO 缓冲提高缓冲性能

在使用 FILE 指针去简单的 fclose 了写指针则会关掉整个套接字，此时虽然读指针没有关闭但已经无法接受信息了，因此这种方式并不是半关闭

# 2. 文件描述符的复制和半关闭

上面问题的原因是，实际上读写指针只是名字上有区分，实际上均指向同一个文件描述符，最好的方式是，在 IO 分离前，就复制一份文件描述符，两个文件描述符均指向同一个套接字，这时再分别让读写指针去分别指向他们即可，但此时关闭其中一个仍然不是半关闭，因为套接字本身并没有被销毁，无法发送 EOF 信息。

解决方法：其实也很简单，就是用 fileno 函数，返回了文件描述符，再用 shutdown 函数即可

以下为复制和分离 IO 以及实现半关闭发送 EOF 的代码

# 17. 优于 select 的 epoll

# ⚠️epoll 仅适用于 Linux，Macos 和 Windows 均无法使用

# 1.epoll 理解及应用

为什么 select 的 io 复用速度慢？

  • 调用 select 后要用 for 循环扫描
  • 每次调用 select 都要向该函数传递监视对象信息，因为有可能有新的客户端连入

每次都需传入监视对象信息对于操作系统是很大的负担，无法通过优化代码解决

其实每次变化时都只变化了某几个项，不需要完整的传入，epoll 函数就是如此

epoll(linux), IOCP(windows)

select 的优点是更通用和兼容，因此在服务器接入者少的情况下仍可作为选择

关于 epoll 的三个重要函数

#include <sys/epoll.h>

//这里展示epoll_event结构体
struct epoll_event
{
	__uint32_t events;
	epoll_data_t data;
	//其中，event有以下类型
	EPOLLIN:读取数据
	EPOLLOUT:输出缓冲为空，可立即发送数据
	EPOLLPRI:收到OOB数据
	EPOLLRDHUP:断开连接或半关闭
	EPOLLERR:发生错误
	EPOLLET:以边缘触发的方式得到事件通知
	EPOLLONESHOT:触发一次后重新设置事件，搭配EPOLL_CTL_MOD
}

//*********************************************************************
int epoll_create(size);
//注意以目前linux的版本，这个size是系统自动分配调整的

//*********************************************************************
int epoll_ctl(int epfd, int op, int fd, struct epoll_event * event);

epoll_ctl(A, EPOLL_CTL_ADD, B, C);
//epoll A中注册文件描述符B，监视参数C中的事件

epoll_ctl(A, EPOLL_CTL_DEL, B, NULL);
//epoll A中删除文件描述符B

//*********************************************************************
int epoll_wait(int epfd, struct epoll_event * event, 
								int maxevents, int timeout))

注意：epoll_event 在使用时需要动态分配内存空间

而 malloc 和 new 都是在堆区开辟内存，需要手动显示释放空间防止内存泄露

{

    socklen_t adr_sz;

    struct epoll_event *ep_events;
    struct epoll_event event;
    int epfd, event_cnt;

    if(argc!=2)
    {
        printf("Usage: %s <port> \\n", argv[0]);
        exit(1);
    }

        {
            puts("epoll wait error");
            break;
        }

        for(i=0;i<event_cnt;i++)
        {
            //client is connecting

            {

# 2. 条件触发和边缘触发

简单来说，条件触发就是只要缓冲里有数据，就一直通知该事件，而边缘触发就是只通知一次，后续即使缓冲里数据没读完，也不会再通知事件

原本默认的就是条件触发，但可以通过设置事件改变，如下

event.events= EPOLLIN||EPOLLET;

注意，边缘触发方式中一定要采用非阻塞 read&write 函数，否则可能引起服务器端长时间停顿，具体方法见下面代码处

# 18. 多线程服务器端的实现

# 1. 线程的概念

本章内容较多，也是因为确实比较重要吧，要加深理解，多加应用

多进程服务器的缺点：

1. 创建进程的开销

2. 进程间的数据交换较为繁琐

3. 频繁的上下文切换，十分消耗资源

上下文切换，需要反复访存 IO，十分消耗时间，保存上下文，更新 PCB，切换上下文之类的

线程创建切换上下文更快，并且最重要的是共享内存资源

进程需要数据区，堆区，栈区

而线程共享数据区，堆区。只有单独的栈区，确保各自的函数执行流

进程相当于在操作系统里创建了各种执行流，而线程相当于在进程里创建了多种执行流

# 2. 线程的创建及运行

    int range1[]={1,100};
    int range2[]={100,1000};

    pthread_create(&id_t1,NULL,thread_summation,(void*)range1);
    pthread_create(&id_t2,NULL,thread_summation,(void*)range2);

    pthread_join(id_t2,NULL);
    printf("Sum: %d\\n",sum);
    return 0;
}

void *thread_summation(void * arg)
{
    int start=((int*)arg)[0];
    int end=((int*)arg)[1];
    while(start<=end)
    {
        printf("Now: + %d \\n",start);
        sum+=start;
        start++;
    }
    return NULL;
}

我修改了范围值，可以看到的是执行顺序是随时可能变化的

这里已经可以看到多线程时对于临界区访问的一些问题了

例举一个最简单的例子，进程中的 num 为 99，线程 1 和线程 2 均让 num+1，实现线程 1 会获取 num，为 99，再完成自增，但如果这时候发生线程切换呢，线程 2 也访问 num 发现是 99，最终无论谁写回，都是 100，而正确答案应该是 101

这其实就是要保证对临界区资源的访问要一气呵成，类似原语，操作系统里学的方法是 pv 操作，保证同步和互斥，mutex

要注意预防死锁

同步与互斥问题在操作系统中已学习了不少，接下来是互斥和同步的代码部分

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pthread.h>

#define NUM_THREAD 100
void *thread_inc(void *arg);
void *thread_des(void *arg);

long long num=0;
pthread_mutex_t mutex;

int main(int argc, char const *argv[])
{
    pthread_t thread_id[NUM_THREAD];
    int i;

    //创建互斥信号量
    pthread_mutex_init(&mutex, NULL);

    for(i=0;i<NUM_THREAD;i++)
    {
        if(i%2==0)
        {
            pthread_create(&(thread_id[i]),NULL,thread_inc,NULL);
        }
        else
        {
            pthread_create(&(thread_id[i]),NULL,thread_des,NULL);
        }
    }
    for(i=0;i<NUM_THREAD;i++)
     pthread_join(thread_id[i],NULL);
    printf("result: %lld \\n",num);
    pthread_mutex_destroy(&mutex);
    return 0;
}

void *thread_inc(void *arg)
{
    int i;
    pthread_mutex_lock(&mutex);
    for(i=0;i<50000;i++)
    {
        num++;
    }
    pthread_mutex_unlock(&mutex);
    return NULL;
}

void *thread_des(void *arg)
{
    int i;
    pthread_mutex_lock(&mutex);
    for(i=0;i<50000;i++)
    {
        num--;
    }
    pthread_mutex_unlock(&mutex);
    return NULL;
}

#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>

void * read(void * arg);
void * accu(void * arg);
static sem_t sem_one;
static sem_t sem_two;
static int num;

int main(int argc, char const *argv[])
{
    pthread_t id_t1,id_t2;
    sem_init(&sem_one, 0 ,0);
    sem_init(&sem_two, 0, 1);

    pthread_create(&id_t1,NULL,read,NULL);
    pthread_create(&id_t2,NULL,accu,NULL);

    pthread_join(id_t1,NULL);
    pthread_join(id_t2,NULL);   

    sem_destroy(&sem_one);
    sem_destroy(&sem_two);
    return 0;
}

//确保读入了数据才能累加，确保累加完毕才能继续读入数据，因此两个同步信号量
void * read(void * arg)
{
    int i;
    for(i=0;i<5;i++)
    {
      fputs("Input:",stdout);
      sem_wait(&sem_two);
      scanf("%d",&num);
      sem_post(&sem_one);
    }
    return NULL;
}

void * accu(void * arg)
{
    int i,sum;
    sum=0;
    for(i=0;i<5;i++)
    {
        sem_wait(&sem_one);
        sum+=num;
        sem_post(&sem_two);
    }
    printf("result: %d \\n",sum);
    return NULL;
}

# 2. 多线程并发服务器端的实现

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <sys/socket.h>
#include <pthread.h>

#define BUF_SIZE 100
#define MAX_CLNT 256

void * handle_clnt(void * arg);
void send_msg(char * msg, int len);
void error_handling(char *message);

int clnt_cnt=0;
int clnt_socks[MAX_CLNT];
pthread_mutex_t mutx;
int main(int argc, char const *argv[])
{
    //sock
    int serv_sock,clnt_sock;
    struct sockaddr_in serv_adr, clnt_adr;
    int clnt_adr_sz;
    pthread_t t_id;

    if(argc!=2)
    {
        printf("Usage:%s <port>",argv[0]);
        exit(1);
    }

    pthread_mutex_init(&mutx,NULL);
    serv_sock=socket(PF_INET, SOCK_STREAM, 0);
    memset(&serv_adr,0,sizeof(serv_adr));
    serv_adr.sin_family=AF_INET;
    serv_adr.sin_addr.s_addr=htonl(INADDR_ANY);
    serv_adr.sin_port=htons(atoi(argv[1]));

    if(bind(serv_sock, (struct sockaddr*)&serv_adr, sizeof(serv_adr))==-1)
        error_handling("bind error");
    if(listen(serv_sock,5)==-1)
        error_handling("listen error");
    
    while(1)
    {
        clnt_adr_sz=sizeof(clnt_adr);
        clnt_sock=accept(serv_sock,(struct sockaddr*)&clnt_adr,&clnt_adr_sz);

        //访问临界区资源要一气呵成，不可同时操作修改
        pthread_mutex_lock(&mutx);
        clnt_socks[clnt_cnt++]=clnt_sock;
        pthread_mutex_unlock(&mutx);

        pthread_create(&t_id,NULL,handle_clnt,(void *)&clnt_sock);
        pthread_detach(t_id);
        printf("Connected client IP:%s \\n", inet_ntoa(clnt_adr.sin_addr));
    }
    close(serv_sock);
    return 0;
}

void * handle_clnt(void * arg)
{
    int clnt_sock=*((int *)arg);
    int str_len=0,i;
    char msg[BUF_SIZE];

    //重复更新信息给所有客户端,当str_len为0说明对方断开连接
    while((str_len=read(clnt_sock,msg,sizeof(msg)))!=0)
        send_msg(msg, str_len);
    //下面是断开连接后对原线性表的维护
    //时刻关注各个相关参数的处理
    pthread_mutex_lock(&mutx);
    for(i=0;i<clnt_cnt;i++)
    {
        if(clnt_sock==clnt_socks[i])
        {
            while(i++<clnt_cnt-1)
                {
                    clnt_socks[i]=clnt_socks[i+1];
                }
            break;
        }
    }
    clnt_cnt--;
    pthread_mutex_unlock(&mutx);
    close(clnt_sock);
    return NULL;
}
void send_msg(char * msg, int len)
{
    //send to all clients
    int i;
    pthread_mutex_lock(&mutx);
    for(i=0;i<clnt_cnt;i++)
    {
        write(clnt_socks[i],msg,len);
    }
    pthread_mutex_unlock(&mutx);
}
void error_handling(char *message)
{
    fputs(message,stderr);
    fputc('\\n',stderr);
    exit(1);
}
