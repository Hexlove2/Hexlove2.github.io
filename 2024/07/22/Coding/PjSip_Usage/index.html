Example:

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <pjsua-lib/pjsua.h>

#define THIS_FILE "APP"

//Callback Function for calling
void on_incoming_call(pjsua_acc_id acc_id, pjsua_call_id call_id, pjsip_rx_data *rdata)
{
    printf("\n");
    printf("-------------------------------------\n");
    printf("|           on_incoming_call        |\n");
    printf("-------------------------------------\n");


    pjsua_call_info ci;

    PJ_UNUSED_ARG(acc_id);
    PJ_UNUSED_ARG(rdata);

    //Retrieve the detailed information about the incoming call
    pjsua_call_get_info(call_id, &ci);

    PJ_LOG(3, (THIS_FILE, "Incoming call from %.*s", (int)ci.remote_info.slen, ci.remote_info.ptr));

    //Show the information from the call
    printf("\n Incoming call from: %.*s \n", (int)ci.remote_info.slen, ci.remote_info.ptr);
    printf("--------------------------------\n");
    printf("|  Enter A to Accept The Call  |\n");
    printf("--------------------------------\n");

    //Automatically Answer The Call(Optional)
    //pjsua_call_answer(call_id, 200, NULL, NULL);

}


//Callback Function for call state changes
void on_call_state(pjsua_call_id call_id, pjsip_event *e)
{
    printf("\n");
    printf("-------------------------------------\n");
    printf("|            on_call_state          |\n");
    printf("-------------------------------------\n");
    pjsua_call_info ci;

    PJ_UNUSED_ARG(e);

    //Retrieve the detailed information about the incoming call
    pjsua_call_get_info(call_id, &ci);

    PJ_LOG(3, (THIS_FILE, "Call: %d  State %.*s", call_id, (int)ci.state_text.slen, ci.state_text.ptr));

}


//Callback Function for media state of call changes
void on_call_media_state(pjsua_call_id call_id)
{

    printf("\n");
    printf("-------------------------------------\n");
    printf("|         on_call_media_state       |\n");
    printf("-------------------------------------\n");

    pjsua_call_info ci;

    //Retrieve the detailed information about the incoming call
    pjsua_call_get_info(call_id, &ci);

    //Connect to the audio stream if the media state is active
    if(ci.media_status == PJSUA_CALL_MEDIA_ACTIVE)
    {
        pjsua_conf_connect(ci.conf_slot, 0);
        pjsua_conf_connect(0, ci.conf_slot);
    }    
}

//Print Help Information
void print_help() 
{
    printf("Available commands:\n");
    printf(" -------------------------\n");
    printf(" |m| - |  Make new call  |\n");
    printf(" |a| - |   Answer call   |\n");
    printf(" |h| - |   Hangup call   |\n");
    printf(" |q| - |     Quit        |\n");
    printf(" -------------------------\n");
}


int main(int argc, char const *argv[])
{
    //Port Extract
    if(argc < 2)
    {
        printf("Usage: %s <port> \n", argv[0]);
        return 1;
    }

    unsigned short port = (unsigned short)atoi(argv[1]);


    printf("-------------------------------------------------------- \n");
    printf("PJSIP START \n");
    printf("-------------------------------------------------------- \n");

    //Data Structure
    pjsua_config                    cfg;
    pjsua_acc_id                    acc_id;
    pjsua_logging_config            log_cfg;
    pjsua_acc_config                acc_cfg;
    pjsua_transport_config          transport_cfg;
    pj_status_t                     status;

    //Create PJSUA Instance
    status = pjsua_create();
    if(status != PJ_SUCCESS)
    {
        return 1;
    }

    //PJSUA DEFAULT SET
    pjsua_config_default(&cfg);
    cfg.cb.on_incoming_call     = &on_incoming_call;
    cfg.cb.on_call_state        = &on_call_state;
    cfg.cb.on_call_media_state  = &on_call_media_state;

    //PJSUA LOGGING SET
    pjsua_logging_config_default(&log_cfg);
    log_cfg.console_level       = 4;

    //INITIALIZE PJSUA
    status = pjsua_init(&cfg, &log_cfg, NULL);
    if(status != PJ_SUCCESS)
    {
        pjsua_destroy();
        return 1;
    }

    //PJSUA TRANSPORT SET
    pjsua_transport_config_default(&transport_cfg);
    transport_cfg.port = port;
    status = pjsua_transport_create(PJSIP_TRANSPORT_UDP, &transport_cfg , NULL);
    if(status != PJ_SUCCESS)
    {
        pjsua_destroy();
        return 1;
    }    

    //SET SIP ACCOUNT
    pjsua_acc_config_default(&acc_cfg);
    acc_cfg.id              = pj_str("sip:11702@192.168.12.36");
    acc_cfg.reg_uri         = pj_str("sip:192.168.12.36");
    acc_cfg.cred_count      = 1;

    acc_cfg.cred_info[0].realm     = pj_str("*");
    acc_cfg.cred_info[0].scheme    = pj_str("digest");
    acc_cfg.cred_info[0].username  = pj_str("11702");
    acc_cfg.cred_info[0].data_type = PJSIP_CRED_DATA_PLAIN_PASSWD;
    acc_cfg.cred_info[0].data      = pj_str("1234");

    //ADD SIP ACCOUNT
    status = pjsua_acc_add(&acc_cfg, PJ_TRUE, &acc_id);
    if(status != PJ_SUCCESS)
    {
        pjsua_destroy();
        return 1;
    }

    //START PJSUA
    status = pjsua_start();
    if(status != PJ_SUCCESS)
    {
        pjsua_destroy();
        return 1;
    }   

    //PRINT PROJECT DIRECTORY
    char cwd[1024];
    if((getcwd(cwd, sizeof(cwd))) != NULL)
        printf("CURRENT WORKING DIRECTORY : %s" , cwd);
    else
        perror("getcwd error");


    print_help();

    char input[10];
    //START
    while(1)
    {
        printf("----------------------------------------------");
        if(fgets(input, sizeof(input), stdin) == NULL)
            break;
        
        //Make a new call
        if(input[0] == 'M' || input[0] == 'm')
        {
            char uri[64];
            printf("Enter Destination SIP URI: ");
            if(fgets(uri, sizeof(uri), stdin) == NULL)  break;

            uri[strcspn(uri, "\n")] = '\0';

            pj_str_t         dest_uri = pj_str(uri);
            pjsua_call_id                       call_id;

            status = pjsua_call_make_call(acc_id, &dest_uri, 0, NULL, NULL, &call_id);
            if(status != PJ_SUCCESS)
            {
                PJ_LOG(1, (THIS_FILE, "Error making call"));
            }

        }

        //Accept the call
        else if(input[0] == 'A' || input[0] == 'a')
        {
            pjsua_call_info ci;

            for(pjsua_call_id i = 0; i < PJSUA_MAX_CALLS; i++)
            {
                if(pjsua_call_is_active(i))
                {
                    pjsua_call_get_info(i, &ci);
                    if(ci.role == PJSIP_ROLE_UAS && ci.state == PJSIP_INV_STATE_INCOMING)
                    {
                        pjsua_call_answer(i, 200, NULL, NULL);
                        break;
                    }
                }
            }
        }

        //Hang up all the call
        else if(input[0] == 'H' || input[0] == 'h')
        {
            pjsua_call_hangup_all();
        }

        //Quit
        else if(input[0] == 'Q' || input[0] == 'q')
        {
            break;
        }

        else
        {
            print_help();
        }
    }

    pjsua_destroy();
    return 0;
}

# 4. Data Structure

# 1. pjsua_acc_id

pjsua_acc_id

This is a type defined in pjsip that represents the ID of a sip account, here is the usage:

	pjsua_acc_id acc_id;
//Add SIP Account
  status = pjsua_acc_add(&acc_cfg, PJ_TRUE, &acc_id);
//Call
  status = pjsua_call_make_call(acc_id, &dest_uri, 0, NULL, NULL, &call_id);

Alright, I know its a little confusing for you, lets talk about status.

# 2. pj_status_t

pj_status_t status = pjsua_create();

Its used to represent the return status of PJSIP functions, here are some common values.

	PJ_SUCCESS: Indicates that the function call was successful. It is usually defined as 0.
	PJ_EUNKNOWN: An unknown error occurred.
	PJ_ENOMEM: Out of memory error.
	PJ_EINVAL: Invalid argument.
	PJ_ECANCELLED: Operation cancelled.

Usage:

if(status != PJ_SUCCESS)
{
  return 1;
}

status = pjsua_init(&cfg, &log_cfg, NULL);
status = pjsua_transport_create(PJSIP_TRANSPORT_UDP, &transport_cfg, NULL);
status = pjsua_acc_add(&acc_cfg, PJ_TRUE, &acc_id);
status = pjsua_start();
status = pjsua_call_make_call(acc_id, &dest_uri, 0, NULL, NULL, &call_id);

# 3. pjsua_config

pjsua_config cfg

Its a structure of PJSIP that holds the configuration settings for the PJSUA library. This structure is used to initialize various parameters and callbacks that PJSUA will use during its operation.

pjsua_config_default(&cfg);
cfg.cb.on_incoming_call = &on_incoming_call;
cfg.cb.on_call_state = &on_call_state;
cfg.cb.on_call_media_state = &on_call_media_state;

# 4. pjsua_logging_config

pjsua_logging_config log_cfg;

This structure allows you to control how and where log messages are output, as well as the verbosity level of the logging.

  pjsua_logging_config_default(&log_cfg);
  log_cfg.console_level = 4;
status = pjsua_init(&cfg, &log_cfg, NULL);

# 5. pjsua_transport_config

pjsua_transport_config transport_cfg;

Transports in PJSIP are used to send and receive SIP messages over the network. The most common types of transport are UDP, TCP, TLS.

pjsua_transport_config_default(&transport_cfg);
transport_cfg.port = port;  // SIP
status = pjsua_transport_create(PJSIP_TRANSPORT_UDP, &transport_cfg, NULL);

# 6. pjsua_acc_config

pjsua_acc_config acc_cfg;

This structure is used to set up various parameters for the SIP account, such as the accounts SIP URI, registration URI, authentication credentials, and other settings.

// SIP
pjsua_acc_config_default(&acc_cfg);
acc_cfg.id = pj_str("sip:11702@192.168.12.36");  // 
acc_cfg.reg_uri = pj_str("sip:192.168.12.36");  // SIPURI
acc_cfg.cred_count = 1;
acc_cfg.cred_info[0].realm = pj_str("*");  // 
acc_cfg.cred_info[0].scheme = pj_str("digest");
acc_cfg.cred_info[0].username = pj_str("11702");  // 
acc_cfg.cred_info[0].data_type = PJSIP_CRED_DATA_PLAIN_PASSWD;
acc_cfg.cred_info[0].data = pj_str("1234");  // 

// SIP
status = pjsua_acc_add(&acc_cfg, PJ_TRUE, &acc_id);

# 7. pjsua_call_info

pjsua_call_info ci;

Its a structure of PJSUA that contains detailed information about a call. Its used to retrieve various details about a call, such as its current state, media status, remote and local identities, and other relevant information. Its typically used in callback functions to handle events related to calls.

// 
pjsua_call_get_info(call_id, &ci);
PJ_LOG(3, (THIS_FILE, "Incoming call from %.*s!!",
           (int)ci.remote_info.slen, ci.remote_info.ptr));

// 
printf("\nIncoming call from %.*s\n", (int)ci.remote_info.slen, ci.remote_info.ptr);
